# defining a function
# function w/out parameters
def hello_world
    print "Hello World!"
end

# function w/ parameters
def hello(name)
    print "Hello " + name
end

# splat arguments
def greet(*friends)
    friends.each { |friend| puts "Hi, #{friend}!" }
end

# calling a function
hello_world # Hello World!

# calling a function w/ parameters
hello("Bob") # Hello Bob

greet("Bob", "Mila", "John")
# Hi Bob!
# Hi Mila!
# Hi John!
